# Batch Call Examples
#
# Usage: cat examples/batch-calls.txt | mcpx call
#
# Format: server.tool({ arg1: "value", arg2: 123 })
# - Supports JSON5 syntax (trailing commas, comments, unquoted keys)
# - Empty lines and comments (# or //) are ignored
# - Output is in TOON format (compact, LLM-friendly)

# Example 1: Linear issue management
linear.list_issues({ limit: 5, status: "in-progress" })
linear.get_issue({ id: "ENG-123" })

# Example 2: GitHub operations
github.search_repos({ query: "mcp", limit: 10 })
github.get_repo({ owner: "AIGC-Hackers", repo: "mcpx" })

# Example 3: Multiple servers in one batch
notion.query_database({ databaseId: "abc123", limit: 20 })
slack.send_message({ channel: "#general", text: "Batch job completed" })

# Example 4: Complex arguments with JSON5
linear.create_issue({
  title: "Fix batch processing bug",
  description: "Found during testing",
  teamId: "team-123",
  priority: 2,
  labels: ["bug", "high-priority"]  // trailing comma is OK!
})
